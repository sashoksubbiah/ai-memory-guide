The Multi-Agent Permission Model

Policies

Access policy table

permission.py
WRITE_POLICY = {
    ("researcher", "research_note", "private"):  "ALLOW",
    ("researcher", "research_note", "project"):  "ALLOW",
    ("researcher", "research_note", "user"):     "DENY",
    ("coder",      "code_pattern",  "project"):  "ALLOW",
    ("coder",      "code_pattern",  "org"):      "DENY",
    ("support",    "customer_ep",   "user"):     "ALLOW",
}

def can_write(agent_role, content_kind, target_scope):
    return WRITE_POLICY.get((agent_role, content_kind, target_scope), "DENY") == "ALLOW"
💡 Deny by default

Any (agent, content, scope) triple not explicitly in the policy table defaults to DENY. This prevents accidental over-sharing when new agent types are added.

Implementation

In LangGraph / AutoGen

langraph_scopes.py
# Scoped memory stores
from langgraph.store import MemoryStore

# Per-agent private store
agent_stores = {agent_id: MemoryStore(namespace=("private", agent_id)) for agent_id in agents}

# Shared project store
project_store = MemoryStore(namespace=("project", project_id))

# User profile store
user_store = MemoryStore(namespace=("user", user_id))

# Governed cross-scope write
def project_write(agent_id, content, kind):
    if not can_write(agent_roles[agent_id], kind, "project"):
        raise PermissionError(f"{agent_id} cannot write {kind} to project")
    audit_log(agent_id, content, kind, "project")
    project_store.write(content, author=agent_id)
